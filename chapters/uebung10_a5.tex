\begin{align}

    \end{pmatrix}

    \begin{pmatrix}
        -3y^2 + 3x^2 \\

    \end{pmatrix}
    &= \begin{pmatrix}
        0 \\
        0
    \end{pmatrix} \\
    \begin{pmatrix}
        x^2 \\

    \end{pmatrix}
    &= \begin{pmatrix}
        y^2 \\
        6x
    \end{pmatrix}
    \begin{matrix}
        \\
        \vee y = 0
    \end{matrix} \\
    \begin{pmatrix}
        x^2 \\
        4x^2 - 6x

    =& \begin{pmatrix}

    \end{pmatrix}
    \begin{matrix}
        \\
        \vee y = 0
    \end{matrix} \\
\end{align}

\begin{align}
    4x^2 - 6 x &= 0 \\
    x^2 - \frac{3}{2} x &= 0 \\
    x - \frac{3}{2} &= 0 \vee x = 0 \\
    x &= \frac{3}{2} \vee x = 0
\end{align}

Lokale Extreme liegen bei:
\begin{align}
    (x,y) &= (0,0) \\
    &= (\frac{3}{2}, \frac{3}{2}) \\
    \\
    f(0,0) &= 0 \\
    f(\frac{3}{2}, \frac{3}{2}) &\approx -1.69
\end{align}
